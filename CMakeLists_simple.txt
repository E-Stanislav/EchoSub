cmake_minimum_required(VERSION 3.16)
project(SimpleMediaPlayer VERSION 1.0.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Enable Qt MOC and UIC
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTOUIC ON)

# Qt 6 - только необходимые компоненты
find_package(Qt6 REQUIRED COMPONENTS Core Widgets Multimedia MultimediaWidgets)

# Include directories
include_directories(${CMAKE_SOURCE_DIR}/include)

# Source files для простого плеера
set(SIMPLE_SOURCES
    src/simple_player_test.cpp
    src/core/simplemediaplayer.cpp
)

set(SIMPLE_HEADERS
    include/core/simplemediaplayer.h
)

# Create executable для простого плеера
add_executable(simple_player ${SIMPLE_SOURCES} ${SIMPLE_HEADERS})

# Link libraries - только Qt, без FFmpeg
target_link_libraries(simple_player 
    Qt6::Core 
    Qt6::Widgets 
    Qt6::Multimedia 
    Qt6::MultimediaWidgets
)

# Set output directory
set_target_properties(simple_player PROPERTIES
    RUNTIME_OUTPUT_DIRECTORY ${CMAKE_BINARY_DIR}/bin
)

# Install
install(TARGETS simple_player DESTINATION bin)

# macOS specific settings
if(APPLE)
    set_target_properties(simple_player PROPERTIES
        MACOSX_BUNDLE TRUE
        MACOSX_BUNDLE_INFO_PLIST ${CMAKE_SOURCE_DIR}/Info.plist
    )
endif()

# Print configuration info
message(STATUS "Simple Media Player Configuration:")
message(STATUS "  Qt6 version: ${Qt6_VERSION}")
message(STATUS "  Using built-in Qt multimedia components")
